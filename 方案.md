```typescript
      const { diaries, hasMore } = await api.diaries.getList({ page });
      set((state) => ({

        hasMore,
        page,

      set({ isLoading: false });

      diaries: [diary, ...state.diaries],

      set({
        aiInsight: result.aiInsight,
        emotionScore: result.emotionScore,
        emotionTags: result.emotionTags,

    });
```

```typescript
interface SyncState {

  pendingCount: number;
  lastSyncTime: Date | null;
  
  setOnline: (online: boolean) => void;
  sync: () => Promise<void>;

}

  isOnline: navigator.onLine,
  isSyncing: false,
  pendingCount: 0,
  lastSyncTime: null,

    set({ isOnline: online });
    if (online) {
      get().sync();
    }
  },
  
  sync: async () => {
    set({ isSyncing: true });
    try {
      await syncService.executeSync();
      set({ lastSyncTime: new Date(), pendingCount: 0 });
    } finally {
      set({ isSyncing: false });
    }
  },
  
  addToQueue: (item) => {
```

```typescript
// client/src/utils/eventConflict.ts

interface Event {
  id: string;
  startTime: Date;
  endTime: Date;
}

function checkConflict(
  newEvent: Omit<Event, 'id'>,
  existingEvents: Event[]
): Event | null {
  for (const event of existingEvents) {
    // 检查时间重叠
    const overlap = (
      newEvent.startTime < event.endTime &&
      newEvent.endTime > event.startTime
    );
    
    if (overlap) {
      return event; // 返回冲突的日程
    }
  }
  return null;
}

// 使用示例
const conflict = checkConflict(
  { startTime: new Date('2026-04-29T14:00'), endTime: new Date('2026-04-29T15:00') },
  existingEvents
);

if (conflict) {
  toast.warning(`与"${conflict.title}"时间冲突`);
}
```

```typescript
// client/src/api/client.ts
import axios from 'axios';

  baseURL: import.meta.env.VITE_API_URL,

  (config) => {

      config.headers.Authorization = `Bearer ${token}`;
    }

  },

apiClient.interceptors.response.use(
  (response) => response,
  (error) => {
    if (error.response) {
      switch (error.response.status) {
        case 401:
          // Token过期，刷新Token
          useAuthStore.getState().refreshToken();
          break;
        case 403:
          // 权限不足
          toast.error('权限不足');
          break;

          toast.error('请求的资源不存在');
          break;
        case 429:
          // 请求过于频繁
          toast.error('请求过于频繁，请稍后再试');
          break;
        case 500:
          // 服务器错误
          toast.error('服务器错误，请稍后再试');
          break;
        default:
          toast.error('发生未知错误');
      }
    } else if (error.request) {
      // 网络错误
      toast.error('网络连接失败，已保存到本地');
      // 将请求加入同步队列
      useSyncStore.getState().addToQueue({
        entityType: 'diary',
        entityId: generateId(),
        action: 'create',
        payload: error.config.data,
      });
    }
    return Promise.reject(error);
  }
);
```

```typescript
// server/src/middleware/error.ts
import { FastifyError, FastifyReply, FastifyRequest } from 'fastify';

export const errorHandler = (
  error: FastifyError,
  request: FastifyRequest,
  reply: FastifyReply
) => {
  // 记录错误日志
  request.log.error(error);
  
  // 根据错误类型返回不同响应
  if (error.validation) {
    // 参数校验错误
    return reply.status(400).send({
      error: 'Bad Request',
      message: error.message,
      details: error.validation,
    });
  }
  
  if (error.statusCode === 401) {
    return reply.status(401).send({
      error: 'Unauthorized',
      message: '认证失败，请重新登录',
    });
  }
  
  if (error.statusCode === 404) {
    return reply.status(404).send({
      error: 'Not Found',
      message: '请求的资源不存在',
    });
  }
  
  // 默认500错误
  return reply.status(500).send({
    error: 'Internal Server Error',
    message: process.env.NODE_ENV === 'production' 
      ? '服务器内部错误' 
      : error.message,
  });
};
```

```typescript
// server/src/middleware/auth.ts
import jwt from 'jsonwebtoken';

export const authenticate = async (request: FastifyRequest, reply: FastifyReply) => {
  try {
    const token = request.headers.authorization?.replace('Bearer ', '');
    
    if (!token) {
      return reply.status(401).send({ error: '缺少认证令牌' });
    }
    
    const decoded = jwt.verify(token, process.env.JWT_SECRET!) as JWTPayload;
    request.user = decoded;
  } catch (error) {
    return reply.status(401).send({ error: '认证令牌无效' });
  }
};

  return bcrypt.compare(password, hash);
```

### 17.2 API安全

```typescript
// server/src/middleware/rateLimit.ts
import { FastifyRequest, FastifyReply } from 'fastify';

const rateLimitMap = new Map<string, { count: number; resetTime: number }>();

export const rateLimit = (maxRequests: number, windowMs: number) => {
  return async (request: FastifyRequest, reply: FastifyReply) => {
    const key = request.user?.id || request.ip;
    const now = Date.now();
    
    const record = rateLimitMap.get(key);
    
    if (!record || now > record.resetTime) {
      rateLimitMap.set(key, { count: 1, resetTime: now + windowMs });
      return;
    }
    
    if (record.count >= maxRequests) {
      return reply.status(429).send({
        error: 'Too Many Requests',
        message: '请求过于频繁，请稍后再试',
        retryAfter: Math.ceil((record.resetTime - now) / 1000),
      });
    }
    
    record.count++;
  };
};

// 使用示例
app.get('/api/ai/chat', { preHandler: rateLimit(10, 60000) }, handler);
```
